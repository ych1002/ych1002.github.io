#mysql第二次课
/*
数据库设计的三大范式
第一范式：
每个字段的数据都是不可分割
的最小的数据单元
第二范式
每张表只能描述一件事情
每张表必须有主键
第三范式
非主键字段依赖于主键字段
换句话说，非主键字段与
主键字段要直接相关
注意：设计表时，尽可能遵守规范
但是，有时为了性能，会牺牲
一点规范
Mysql运算符
算术运算符
+ - * /
逻辑运算符
and or !
比较运算符
> < >= <= <> !=
聚合函数
count() 求行数
sum() 求一列的和
avg() 去平均值
max() 求最大值
min() 求最小值
Mysql常用关键字
in 查询在什么什么选项中的数据
where name in(‘张三’,’李四’)
把name等于张三，等于李四的都
查询出来
not in
where name not in(‘张三’,’李四’)
把name不等于张三，不等于李四
的都查询出来
distinct 去重
limit 查询出表的前几行
between…and..在什么什么之间
as
/
SELECT * FROM goods;
INSERT INTO goods(gnam,gprice)
VALUES(‘吹风机’,89);
INSERT INTO goods(gnam,gprice)
VALUES(‘外套’,200);
INSERT INTO goods(gnam,gprice)
VALUES(‘外套’,220);
INSERT INTO goods(gnam,gprice)
VALUES(‘吹风机’,78);
#查询出商品价格大于10的商品
SELECT * FROM goods
WHERE gprice>10;
#查询商品价格不等于10的商品信息
SELECT * FROM goods
WHERE gprice<>10;
#查询商品价格大于10并且小于50的
#商品信息
SELECT * FROM goods WHERE
gprice>10 AND gprice<50;
#查询出商品的数量
#字段名 as 新字段名 给列名
#重命名成新的列名
SELECT COUNT() AS ‘总数量’
FROM goods;
#查询商品的总价格
SELECT SUM(gprice) ‘总价格’
FROM goods;
#查询商品的平均价格
SELECT AVG(gprice) ‘平均价格’
FROM goods;
#查询商品的最高价格
SELECT MAX(gprice) ‘最高价格’
FROM goods;
#查询商品的最低价格
SELECT MIN(gprice) ‘最低价格’
FROM goods;
#查询出商品名称是小熊雨伞，外套的
#都查询出来
SELECT * FROM goods WHERE gnam
IN(‘小熊雨伞’,’外套’);
SELECT * FROM goods WHERE gnam
NOT IN(‘小熊雨伞’,’外套’);
#把商品名字去重
SELECT DISTINCT gnam FROM goods;
#查询goods表的前两行
#limit 0,2 从下标是0的
#行开始取两行
SELECT * FROM goods LIMIT 0,2;
#查询出价格处于10到50之间
#的商品信息
SELECT * FROM goods WHERE
gprice BETWEEN 10 AND 50;
